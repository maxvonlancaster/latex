\input fontinst.sty

\needsfontinstversion{1.328}

\declareencoding{MALVERN A 1.1}{pdcmaa}
\declareencoding{MALVERN B 1.1}{pdcmab}

\installfonts
    \installfamily{T1}{fmv}{}
    \installfont{fmvmq10}{ma55a10,ma55b10,pdcmat1,latin}
	{T1}{T1}{fmv}{m}{n}{10}
\endinstallfonts
